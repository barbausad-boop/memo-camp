async function enregistrerDossier(){ const first_name = document.getElementById("first_name").value.trim(); const last_name = document.getElementById("last_name").value.trim(); const date_naissance = document.getElementById("date_naissance").value.trim(); const lieu_naissance = document.getElementById("lieu_naissance").value.trim(); const phone = document.getElementById("phone").value.trim(); const email = document.getElementById("email").value.trim(); const groupe_scout = document.getElementById("groupe_scout").value.trim(); const district = document.getElementById("district").value.trim(); const fonction = document.getElementById("fonction").value.trim(); const stage = document.getElementById("stage").value.trim(); const branche = document.getElementById("branche").value.trim(); const annee_stage = document.getElementById("annee_stage").value.trim(); const lieu_stage = document.getElementById("lieu_stage").value.trim(); // Vérification champs obligatoires if( !first_name || !last_name || !date_naissance || !lieu_naissance || !phone || !email || !groupe_scout || !district || !fonction || !stage || !branche || !annee_stage || !lieu_stage ){ alert("Tous les champs sont obligatoires"); return; } // Vérification doublon const { data: existingParticipant } = await client .from("Participants") .select("*") .eq("first_name", first_name) .eq("last_name", last_name) .eq("phone", phone); if(existingParticipant && existingParticipant.length > 0){ alert("Participant déjà enregistré"); return; } // Création participant const participant = { first_name: first_name, last_name: last_name, groupe_scout: groupe_scout, district: district, fonction: fonction, date_naissance: date_naissance, lieu_naissance: lieu_naissance, phone: phone, email: email }; const { data, error } = await client .from("Participants") .insert([participant]) .select(); if(error){ alert(error.message); return; } const participantId = data[0].id; // Création formation const formation = { participant_id: participantId, stage: stage, branche: branche, annee_stage: annee_stage, lieu_stage: lieu_stage }; const { error: formationError } = await client .from("Formation") .insert([formation]); if(formationError){ alert(formationError.message); return; } alert("Dossier enregistré"); // Vider le formulaire document.getElementById("first_name").value = ""; document.getElementById("last_name").value = ""; document.getElementById("date_naissance").value = ""; document.getElementById("lieu_naissance").value = ""; document.getElementById("phone").value = ""; document.getElementById("email").value = ""; document.getElementById("groupe_scout").value = ""; document.getElementById("district").value = ""; document.getElementById("fonction").value = ""; document.getElementById("stage").value = ""; document.getElementById("branche").value = ""; document.getElementById("annee_stage").value = ""; document.getElementById("lieu_stage").value = ""; chargerParticipants(); filtrerParAnnee(); }
